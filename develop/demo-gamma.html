Reproducible RNG draws between R and Octave: gamma example


  
library(RcppOctave)

## define a Gamma RNG draw function
o_source(text="
  function x = orngg(a, b, n)
    x = b * randg(a, n, 1);
  end
")

N <- 50000

set.seed(42)

x1 <- c(.O$orngg(9,1,N))

x2 <- c(.O$orngg(4,1,N))

x3 <- c(.O$orngg(1,4,N))

set.seed(42)  # reset RNG

y1 <- rgamma(N,9,1)

y2 <- rgamma(N,4,1)

y3 <- rgamma(N,1,1,4)
